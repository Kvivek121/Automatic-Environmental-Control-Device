49void HD44780_Init(uint8_t rows)
50{
51 dpRows = rows;
52
53 dpBacklight = LCD_BACKLIGHT;
54
55 dpFunction = LCD_4BITMODE | LCD_1LINE | LCD_5x8DOTS;
56
57 if (dpRows > 1)
58 {
59 dpFunction |= LCD_2LINE;
60 }
61 else
62 {
63 dpFunction |= LCD_5x10DOTS;
64 }
65
66 /* Wait for initialization */
67 DelayInit();
68 HAL_Delay(50);
69
70 ExpanderWrite(dpBacklight);
71 HAL_Delay(1000);

82

85

88
89 dpControl = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
90 HD44780_Display();
91 HD44780_Clear();

99 HD44780_CreateSpecialChar(1, special2);

101 HD44780_Home();
102}
