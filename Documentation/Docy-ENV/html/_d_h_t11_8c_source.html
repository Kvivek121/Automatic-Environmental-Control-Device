73uint8_t DHT11_Read (void)
74{
75 uint8_t i,j;
76 for (j=0;j<8;j++)
77 {
78 while (!(HAL_GPIO_ReadPin (DHT11_PORT, DHT11_PIN))); // wait for the pin to go high
79 delay (40); // wait for 40 us
80 if (!(HAL_GPIO_ReadPin (DHT11_PORT, DHT11_PIN))) // if the pin is low
81 {
82 i&= ~(1<<(7-j)); // write 0
83 }
84 else i|= (1<<(7-j)); // if the pin is high, write 1
85 while ((HAL_GPIO_ReadPin (DHT11_PORT, DHT11_PIN))); // wait for the pin to go low
86 }
